terraform {
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
      version = "~> 3.0"
    }
  }

  required_version = ">= 1.3.0"
}

provider "azurerm" {
  features {}

  # Optional: Explicitly specify the subscription ID
  subscription_id = "a6e90f5d-2764-44f1-a13b-fe8c9a5eb6b7"
}

# Variables (can be defined in variables.tf or inline)
variable "resource_groups" {
  description = "A map of resource groups to create"
  type        = map(object({
    location = string
  }))
  default = {
    "rg-example-eastus" = { location = "East US" }
    "rg-example-westeurope" = { location = "West Europe" }
  }
}

# Create Resource Groups
resource "azurerm_resource_group" "rg" {
  for_each = var.resource_groups

  name     = each.key
  location = each.value.location
}
